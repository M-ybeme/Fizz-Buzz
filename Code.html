function getValues() {
    let startValue = document.getElementById("startValue").value;
    let endValue = document.getElementById("endValue").value;

    ///parese ints///
    startValue = parseInt(startValue);
    endValue = parseInt(endValue);
    if (Number.isInteger(startValue) && Number.isInteger(endValue)) {
        let numbers = generateNumbers(startValue, endValue)
        displayValues(numbers);
    } else {
        swal.fire({
            backdrop: false,
            title: 'Oops',
            text: "You can only use numbers",
        });
    }
}

//buisness logic function
function generateNumbers(start, stop) {
    let numbers = [];
    for (let i = start; i <= stop; i++) {
        
        if ([i] % 3 == 0 && [i] % 5 == 0) {
            numbers.push("FizzBuzz");
        } else if ([i] % 3 == 0) {
            numbers.push("Fizz")
        } else if ([i] % 5 == 0) {
            numbers.push("Buzz");
        } else {
            numbers.push(i);
        }
    }
    return numbers;
}

The Code is comprised of three parts.

"getValue"

This is the controller function. It retreaves the inputs from the html document. It then calls on the other functions to use these variables.

"generateNumbers"

Uses the values that have been passed into it and determins if they are divisable by 5 and/or 3 by using an if-else statement. Then if those numbers are divisable they are then replaced with the words, "Fizz", "Buzz", or "FizzBuzz". It then returns that into the display fuction.

"displayValues"

The display function uses an if-else statement to asign css classes to the numbers bassed on the same process that determined if they were dividable by 3 and/or 5.
